display_idx = 0
alphas = [5, 10, 20]
betas = [1, 2, 4]
plt.figure(figsize=(15, 15))
plt.gray()
for alpha in alphas:
    for beta in betas:
        customers = 1 + 50
        dishes = 0
        display_idx += 1
        Z = np.zeros((customers, 1000), dtype=np.int8)
        for n in range(1, customers):
            if n == 1:
                p = np.random.poisson(alpha)
                dishes += p
                for i in range(p):
                    Z[n, i] = 1
            else:
                for i in range(dishes):
                    mk = np.sum(Z[:n, i])
                    if np.random.rand(1) < float(mk) / (beta + n - 1):
                        Z[n, i] = 1
                    else:
                        Z[n, i] = 0
                p = np.random.poisson(alpha * beta / (beta + n - 1))
                for i in range(p):
                    Z[n, dishes + i] = 1
                dishes += p
        ax = plt.subplot(len(alphas), len(betas), display_idx)
        ax.matshow(Z[1:, :dishes])
        ax.set_xlabel('Dishes')
        ax.set_ylabel('Customers')
        plt.title('Alpha = ' + str(alpha) + '  Beta = ' + str(beta))
        ax.set_yticklabels([])
        ax.set_xticklabels([])
plt.show();
